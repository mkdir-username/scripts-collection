```python
    - –£–º–Ω–∞—è –æ–±—Ä–∞–±–æ—Ç–∫–∞ Jinja2 –≤–Ω—É—Ç—Ä–∏ —Å—Ç—Ä–æ–∫

    - –î–µ—Ç–∞–ª—å–Ω–∞—è –¥–∏–∞–≥–Ω–æ—Å—Ç–∏–∫–∞ –æ—à–∏–±–æ–∫

    # –®–ê–ì 1: –û–±—Ä–∞–±–æ—Ç–∫–∞ –∏–º–ø–æ—Ä—Ç–æ–≤ —á–µ—Ä–µ–∑ –∫–æ–º–º–µ–Ω—Ç–∞—Ä–∏–∏
    if source_file:

            content, source_file.parent

            logger.info(f"   üì• –ò–º–ø–æ—Ä—Ç—ã –æ–±—Ä–∞–±–æ—Ç–∞–Ω—ã: {import_count}")
    else:
        # –£–¥–∞–ª–µ–Ω–∏–µ –∫–æ–º–º–µ–Ω—Ç–∞—Ä–∏–µ–≤ –¥–ª—è —Ñ–∞–π–ª–æ–≤ –±–µ–∑ source
        content = re.sub(r'(?:^|\s)//[^\n]*', '', content, flags=re.MULTILINE)
        content = re.sub(r'/\*.*?\*/', '', content, flags=re.DOTALL)

    cleaned = content

    # –®–ê–ì 2: –£–º–Ω–∞—è –∑–∞–º–µ–Ω–∞ Jinja2 –±–ª–æ–∫–æ–≤
    # –ü–∞—Ç—Ç–µ—Ä–Ω—ã –≤ –ø–æ—Ä—è–¥–∫–µ –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç–∞ (–æ—Ç —Å–ø–µ—Ü–∏—Ñ–∏—á–Ω—ã—Ö –∫ –æ–±—â–∏–º)
    patterns = [
        # If –±–ª–æ–∫–∏ –≤–Ω—É—Ç—Ä–∏ —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
        (r'"[^"]*\{%\s*if\s+[^%]+%\}[^{]*\{%\s*endif\s*%\}[^"]*"',
         lambda m: self._replace_jinja_in_string(m.group(), 'IF'), 'IF –≤ —Å—Ç—Ä–æ–∫–∞—Ö'),

        # For –±–ª–æ–∫–∏ –≤–Ω—É—Ç—Ä–∏ —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
        (r'"[^"]*\{%\s*for\s+[^%]+%\}[^{]*\{%\s*endfor\s*%\}[^"]*"',
         lambda m: self._replace_jinja_in_string(m.group(), 'FOR'), 'FOR –≤ —Å—Ç—Ä–æ–∫–∞—Ö'),

        # If –±–ª–æ–∫–∏ –≤–Ω–µ —Å—Ç—Ä–æ–∫
        (r'\{%\s*if\s+[^%]+%\}.*?\{%\s*endif\s*%\}',
         '__IF_BLOCK__', 'IF –±–ª–æ–∫–∏'),

        # For –±–ª–æ–∫–∏ –≤–Ω–µ —Å—Ç—Ä–æ–∫
        (r'\{%\s*for\s+[^%]+%\}.*?\{%\s*endfor\s*%\}',

        (r'\{%\s*set\s+[^%]+%\}', '', 'SET –¥–∏—Ä–µ–∫—Ç–∏–≤—ã'),

        (r'\{\{[^}]+\}\}', 'null', '–ü–µ—Ä–µ–º–µ–Ω–Ω—ã–µ'),

        # –û—Å—Ç–∞–ª—å–Ω—ã–µ Jinja2 —Ç–µ–≥–∏
        (r'\{%[^}]+%\}', '', '–ü—Ä–æ—á–∏–µ —Ç–µ–≥–∏'),
    ]

        if callable(replacement):

            matches = list(re.finditer(pattern, cleaned, re.DOTALL))

                cleaned = re.sub(pattern, replacement, cleaned, flags=re.DOTALL)
        else:
            # –î–ª—è —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –∑–∞–º–µ–Ω
            matches = list(re.finditer(pattern, cleaned, re.DOTALL))
            if matches:
                logger.debug(f"   üîç –ó–∞–º–µ–Ω–∞: {description} ({len(matches)} –Ω–∞–π–¥–µ–Ω–æ)")
                for match in reversed(matches):
                    counter += 1
                    key = f"__{description.upper().replace(' ', '_')}_{counter}__"
                    replacements[key] = match.group()
                    cleaned = cleaned[:match.start()] + replacement + cleaned[match.end():]

    # –®–ê–ì 3: –ë–∞–∑–æ–≤–∞—è –æ—á–∏—Å—Ç–∫–∞ —Å–∏–Ω—Ç–∞–∫—Å–∏—Å–∞
    # –£–¥–∞–ª–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã—Ö –∑–∞–ø—è—Ç—ã—Ö
    while ',,' in cleaned:
        cleaned = cleaned.replace(',,', ',')

    # –£–¥–∞–ª–µ–Ω–∏–µ trailing –∑–∞–ø—è—Ç—ã—Ö
    cleaned = re.sub(r',\s*\]', ']', cleaned)
    cleaned = re.sub(r',\s*\}', '}', cleaned)

    # –£–¥–∞–ª–µ–Ω–∏–µ leading –∑–∞–ø—è—Ç—ã—Ö
    cleaned = re.sub(r'\[\s*,', '[', cleaned)
    cleaned = re.sub(r'\{\s*,', '{', cleaned)

    # –ò—Å–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –ø—É—Å—Ç—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
    cleaned = re.sub(r',\s*:', ':', cleaned)
    cleaned = re.sub(r':\s*,', ': null,', cleaned)
    cleaned = re.sub(r':\s*\}', ': null}', cleaned)
    cleaned = re.sub(r':\s*\]', ': null]', cleaned)

    # –®–ê–ì 4: –ü—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω–∞—è –≤–∞–ª–∏–¥–∞—Ü–∏—è
    try:
        json.loads(cleaned)
        logger.debug("   ‚úÖ –ü—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω–∞—è –≤–∞–ª–∏–¥–∞—Ü–∏—è JSON —É—Å–ø–µ—à–Ω–∞")
    except json.JSONDecodeError as e:
        logger.warning(f"   ‚ö†Ô∏è –ü—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã–π JSON –Ω–µ–≤–∞–ª–∏–¥–µ–Ω (—Å—Ç—Ä–æ–∫–∞ {e.lineno}): {e.msg}")

        # Smart —Ä–µ–∂–∏–º - –∞–≤—Ç–æ–∏—Å–ø—Ä–∞–≤–ª–µ–Ω–∏–µ
        if self.smart_mode and self.json_fixer:
            cleaned, fixes = self.json_fixer.fix_json(cleaned)
            if fixes:
                logger.info(f"   üß† Smart –∏—Å–ø—Ä–∞–≤–ª–µ–Ω–∏—è: {', '.join(fixes)}")

                # –ü–æ–≤—Ç–æ—Ä–Ω–∞—è –≤–∞–ª–∏–¥–∞—Ü–∏—è
                try:
                    json.loads(cleaned)
                    logger.info("   ‚úÖ –ê–≤—Ç–æ–∏—Å–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —É—Å–ø–µ—à–Ω–æ")
                except json.JSONDecodeError as e2:
                    logger.error(f"   ‚ùå –ê–≤—Ç–æ–∏—Å–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –Ω–µ –ø–æ–º–æ–≥–ª–æ: {e2.msg}")

                    # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –¥–ª—è debug
                    if self.debug:
                        debug_path = Path(f"/tmp/debug_cleaned_{int(time.time())}.json")
                        with open(debug_path, 'w') as f:
                            f.write(cleaned)
                        logger.error(f"   üìù Debug —Ñ–∞–π–ª: {debug_path}")

    return cleaned, replacements, imported_files


def _replace_jinja_in_string(self, string_value: str, block_type: str) -> str:
    """
    –ó–∞–º–µ–Ω–∞ Jinja2 –≤–Ω—É—Ç—Ä–∏ —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π —Å —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ–º —Å—Ç—Ä—É–∫—Ç—É—Ä—ã

    –ù–û–í–û–ï –≤ v3.5.0:
    - –ö–æ—Ä—Ä–µ–∫—Ç–Ω–∞—è –æ–±—Ä–∞–±–æ—Ç–∫–∞ –≤–ª–æ–∂–µ–Ω–Ω—ã—Ö –±–ª–æ–∫–æ–≤
    - –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –∫–∞–≤—ã—á–µ–∫
    - Fallback –Ω–∞ null –¥–ª—è –ø—É—Å—Ç—ã—Ö —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤

    Args:
        string_value: –°—Ç—Ä–æ–∫–æ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —Å Jinja2 (–Ω–∞–ø—Ä–∏–º–µ—Ä, "text {% if ... %}value{% endif %}")
        block_type: –¢–∏–ø –±–ª–æ–∫–∞ (IF, FOR)

    Returns:
        –û—á–∏—â–µ–Ω–Ω–æ–µ —Å—Ç—Ä–æ–∫–æ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –∏–ª–∏ "null"
    """
    # –£–¥–∞–ª—è–µ–º –≤—Å–µ Jinja2 –∫–æ–Ω—Å—Ç—Ä—É–∫—Ü–∏–∏
    cleaned = re.sub(r'\{%.*?%\}', '', string_value, flags=re.DOTALL)
    cleaned = re.sub(r'\{\{.*?\}\}', '', cleaned)

    # –£–±–∏—Ä–∞–µ–º –ª–∏—à–Ω–∏–µ –ø—Ä–æ–±–µ–ª—ã
    cleaned = re.sub(r'\s+', ' ', cleaned).strip()

    # –ï—Å–ª–∏ —Å—Ç—Ä–æ–∫–∞ —Å—Ç–∞–ª–∞ –ø—É—Å—Ç–æ–π –∏–ª–∏ —Ç–æ–ª—å–∫–æ –∫–∞–≤—ã—á–∫–∏
    if cleaned in ['""', "''", '', '" "', "' '"]:
        logger.debug(f"   üîÑ {block_type} –±–ª–æ–∫ ‚Üí null (–ø—É—Å—Ç–æ–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç)")
        return 'null'

    if not (cleaned.startswith('"') or cleaned.startswith("'")):
        cleaned = f'"{cleaned}"'

    logger.debug(f"   üîÑ {block_type} –±–ª–æ–∫ ‚Üí {cleaned}")
    return cleaned
```
